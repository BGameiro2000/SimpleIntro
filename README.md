# Get Started

```bash
hugo new site <site-name>
cd <site-name>
git submodule add https://github.com/gangjun06/hugo-theme-simple-intro themes/simple-intro
# Apply config
hugo serve -D
```

config.toml
```toml
baseURL = "/"
languageCode = "en-us"
theme = "simple-intro"
title = "Title"

[params]
    mainbg = "backroundImage Path"
    name = "Your Name"
    mainTitle = "Header Title"
    mainText = "Header Text"

[menus]
    [[menu.main]]
        identifier = "about"
        name = "About"
        url = "#about"
    [[menu.main]]
        identifier = "skills"
        name = "Skills"
        url = "#skills"
    [[menu.main]]
        identifier = "projects"
        name = "Projects"
        url = "#projects"
    
```
